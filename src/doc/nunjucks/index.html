Tweakpane is a JavaScript library that provides a compact user interface for fine-tuning/monitoring values. It's originally inspired by the popular library dat.GUI.

Features

  • Super easy to use
  • Clean and simple interface
  • Mobile-friendly
  • Well-documented

<script>
var sketch = new Sketch();

var pane = Tweakpane();

var f1 = pane.addFolder('Appearance');
f1.add(sketch, 'neta', {
  list: ['hotate', 'maguro', 'salmon']
});

var f2 = pane.addFolder('Physics');
f2.add(sketch, 'gravity', {min: 0.1, max: 1.0});
f2.add(sketch, 'airResistance', {min: 0.01, max: 0.1});
f2.add(sketch, 'restitution', {min: 0.0, max: 0.9});
f2.addSeparator();
f2.monitor(sketch.getNeta(), 'z', {
  label: 'netaAlt',
  count: 140,
  graph: true,
  min: 0,
  max: 100
});
f2.addSeparator();
f2.addButton('Pop!').on('click', function() {
  sketch.pop();
});
</script>
